<schema> ::= "{" ( <quote> type_name <quote> ":" <type> )* "}" 

<type> ::= <array_type> | <object_type | <ref_type>

<array_type> ::= "[" <type> "]"

<object_type> ::=
	"{"
		(     <json_string> ":" <type> )?
		( "," <json_string> ":" <type> )*
	"}"


<ref_type> ::= <quote> (<builtin_type> | <type_name>) <quote>

<builtin_type> ::= "string" | "int" | "long" | "decimal" | "double" | "bool" | "raw"

<quote> ::= "\""

<type_name> ::= (letter | number | underscore)+
